contract MASIAToken {
    // ======================
    // TOKEN INFORMATION
    // ======================
    string public constant name = "MASIA";
    string public constant symbol = "MSA";
    uint8 public constant decimals = 18;
    uint256 public totalSupply;
    
    // ======================
    // CORE STORAGE
    // ======================
    mapping(address => uint256) private _balances;
    mapping(address => mapping(address => uint256)) private _allowances;
    mapping(address => VestingSchedule) public vestingSchedules;
    mapping(address => bool) public isVestingRecipient;
    
    // ======================
    // VESTING STRUCTURES
    // ======================
    struct VestingSchedule {
        uint256 startTime;
        uint256 cliffDuration;    // in seconds
        uint256 vestingDuration;  // in seconds
        uint256 totalAmount;
        uint256 releasedAmount;
        bool revocable;
        bool initialized;
    }
    
    // ======================
    // PHASE MANAGEMENT
    // ======================
    struct Phase {
        string name;
        uint256 startTime;
        uint256 endTime;
        uint256 dilutionPercentage; // basis points (100 = 1%)
        bool active;
    }
    
    Phase[] public phases;
    uint256 public currentPhaseIndex = 0;
    address public owner;
    bool public transfersLocked = false;
    
    // ======================
    // EVENTS
    // ======================
    event Transfer(address indexed from, address indexed to, uint256 value);
    event Approval(address indexed owner, address indexed spender, uint256 value);
    event TokensReleased(address indexed holder, uint256 amount);
    event TokensRevoked(address indexed holder, uint256 amount);
    event PhaseAdded(uint256 indexed phaseIndex, string name, uint256 dilutionPercentage);
    event PhaseActivated(uint256 indexed phaseIndex);
    event TokensMinted(address indexed to, uint256 amount, uint256 phaseIndex);
    event TransferLockChanged(bool locked);
    
    // ======================
    // MODIFIERS
    // ======================
    modifier onlyOwner() {
        require(msg.sender == owner, "MASIA: caller is not the owner");
        _;
    }
    
    modifier onlyPhaseManager() {
        require(msg.sender == owner, "MASIA: caller is not authorized");
        _;
    }
    
    modifier transferEnabled() {
        require(!transfersLocked || msg.sender == owner, "MASIA: transfers are locked");
        _;
    }
    
    // ======================
    // CONSTRUCTOR - COMPLETE INITIAL SETUP
    // ======================
    constructor(
        address[] memory _initialHolders,
        uint256[] memory _initialAmounts,
        uint256[] memory _cliffDurations,
        uint256[] memory _vestingDurations,
        bool[] memory _revocableFlags
    ) {
        require(_initialHolders.length == _initialAmounts.length, "MASIA: array length mismatch");
        require(_initialHolders.length == _cliffDurations.length, "MASIA: array length mismatch");
        require(_initialHolders.length == _vestingDurations.length, "MASIA: array length mismatch");
        require(_initialHolders.length == _revocableFlags.length, "MASIA: array length mismatch");
        
        owner = msg.sender;
        totalSupply = 1000000 * 10 ** decimals; // 1,000,000 tokens with 18 decimals
        
        // Initialize all balances to zero first
        for (uint256 i = 0; i < _initialHolders.length; i++) {
            require(_initialHolders[i] != address(0), "MASIA: zero address");
            require(_balances[_initialHolders[i]] == 0, "MASIA: duplicate holder");
        }
        
        // Set up vesting schedules and mint tokens
        uint256 totalAllocated = 0;
        for (uint256 i = 0; i < _initialHolders.length; i++) {
            uint256 amount = _initialAmounts[i] * 10 ** decimals;
            totalAllocated += amount;
            
            // Create vesting schedule
            vestingSchedules[_initialHolders[i]] = VestingSchedule({
                startTime: block.timestamp,
                cliffDuration: _cliffDurations[i] * 30 days, // Convert months to seconds
                vestingDuration: _vestingDurations[i] * 30 days, // Convert months to seconds
                totalAmount: amount,
                releasedAmount: 0,
                revocable: _revocableFlags[i],
                initialized: true
            });
            
            isVestingRecipient[_initialHolders[i]] = true;
            
            // Mint tokens to the contract first (they'll be released via vesting)
            _balances[address(this)] += amount;
        }
        
        require(totalAllocated == totalSupply, "MASIA: total allocation must equal total supply");
        
        // Add default Fibonacci phases
        addPhase("Fase 1: Lanzamiento", 500, 180 days);   // 5% dilution
        addPhase("Fase 2: Crecimiento", 800, 360 days);   // 8% dilution
        addPhase("Fase 3: Escalado", 1300, 540 days);     // 13% dilution
        addPhase("Fase 4: Madurez", 2100, 1095 days);     // 21% dilution
        
        emit Transfer(address(0), address(this), totalSupply);
    }
    
    // ======================
    // VESTING FUNCTIONS - COMPLETE IMPLEMENTATION
    // ======================
    function releaseVestedTokens() public {
        VestingSchedule storage schedule = vestingSchedules[msg.sender];
        require(schedule.initialized, "MASIA: no vesting schedule");
        require(block.timestamp >= schedule.startTime + schedule.cliffDuration, "MASIA: cliff not reached");
        
        uint256 vestedAmount = calculateVestedAmount(msg.sender);
        require(vestedAmount > schedule.releasedAmount, "MASIA: no tokens to release");
        
        uint256 unreleased = vestedAmount - schedule.releasedAmount;
        schedule.releasedAmount = vestedAmount;
        
        // Transfer tokens from contract to holder
        _balances[address(this)] -= unreleased;
        _balances[msg.sender] += unreleased;
        
        emit Transfer(address(this), msg.sender, unreleased);
        emit TokensReleased(msg.sender, unreleased);
    }
    
    function calculateVestedAmount(address holder) public view returns (uint256) {
        VestingSchedule memory schedule = vestingSchedules[holder];
        if (!schedule.initialized) return 0;
        
        // Before cliff - nothing vested
        if (block.timestamp < schedule.startTime + schedule.cliffDuration) {
            return 0;
        }
        
        // After vesting period - everything vested
        if (block.timestamp >= schedule.startTime + schedule.vestingDuration) {
            return schedule.totalAmount;
        }
        
        // During vesting period - linear vesting
        uint256 timeSinceStart = block.timestamp - (schedule.startTime + schedule.cliffDuration);
        uint256 vestingPeriod = schedule.vestingDuration - schedule.cliffDuration;
        
        return (schedule.totalAmount * timeSinceStart) / vestingPeriod;
    }
    
    function revokeVesting(address holder) public onlyOwner {
        VestingSchedule storage schedule = vestingSchedules[holder];
        require(schedule.initialized, "MASIA: no vesting schedule");
        require(schedule.revocable, "MASIA: schedule not revocable");
        require(schedule.releasedAmount < schedule.totalAmount, "MASIA: all tokens already released");
        
        uint256 unreleased = schedule.totalAmount - schedule.releasedAmount;
        schedule.totalAmount = schedule.releasedAmount; // Effectively revokes remaining tokens
        
        // Burn unreleased tokens
        _balances[address(this)] -= unreleased;
        totalSupply -= unreleased;
        
        emit TokensRevoked(holder, unreleased);
        emit Transfer(address(this), address(0), unreleased);
    }
    
    // ======================
    // PHASE MANAGEMENT - COMPLETE IMPLEMENTATION
    // ======================
    function addPhase(string memory name, uint256 dilutionBasisPoints, uint256 duration) public onlyPhaseManager {
        require(dilutionBasisPoints <= 5000, "MASIA: dilution too high"); // Max 50%
        require(duration >= 30 days, "MASIA: duration too short");
        
        uint256 startTime = phases.length == 0 ? block.timestamp : phases[phases.length - 1].endTime;
        
        phases.push(Phase({
            name: name,
            startTime: startTime,
            endTime: startTime + duration,
            dilutionPercentage: dilutionBasisPoints,
            active: false
        }));
        
        emit PhaseAdded(phases.length - 1, name, dilutionBasisPoints);
    }
    
    function activateNextPhase() public onlyPhaseManager {
        require(currentPhaseIndex < phases.length, "MASIA: no more phases");
        require(!phases[currentPhaseIndex].active, "MASIA: current phase already active");
        require(block.timestamp >= phases[currentPhaseIndex].startTime, "MASIA: phase not started yet");
        
        // Deactivate previous phase if exists
        if (currentPhaseIndex > 0) {
            phases[currentPhaseIndex - 1].active = false;
        }
        
        // Activate current phase
        phases[currentPhaseIndex].active = true;
        emit PhaseActivated(currentPhaseIndex);
        
        // Calculate and mint new tokens for this phase
        uint256 dilution = phases[currentPhaseIndex].dilutionPercentage;
        uint256 newTokens = (totalSupply * dilution) / (10000 - dilution);
        
        _mint(owner, newTokens);
        currentPhaseIndex++;
    }
    
    // ======================
    // ERC20 CORE FUNCTIONS
    // ======================
    function transfer(address to, uint256 amount) public transferEnabled returns (bool) {
        _transfer(msg.sender, to, amount);
        return true;
    }
    
    function approve(address spender, uint256 amount) public returns (bool) {
        _approve(msg.sender, spender, amount);
        return true;
    }
    
    function transferFrom(address from, address to, uint256 amount) public transferEnabled returns (bool) {
        _spendAllowance(from, msg.sender, amount);
        _transfer(from, to, amount);
        return true;
    }
    
    function balanceOf(address account) public view returns (uint256) {
        return _balances[account];
    }
    
    function allowance(address owner, address spender) public view returns (uint256) {
        return _allowances[owner][spender];
    }
    
    function _transfer(address from, address to, uint256 amount) internal {
        require(from != address(0), "ERC20: transfer from zero address");
        require(to != address(0), "ERC20: transfer to zero address");
        require(_balances[from] >= amount, "ERC20: insufficient balance");
        
        _balances[from] -= amount;
        _balances[to] += amount;
        
        emit Transfer(from, to, amount);
    }
    
    function _approve(address owner, address spender, uint256 amount) internal {
        require(owner != address(0), "ERC20: approve from zero address");
        require(spender != address(0), "ERC20: approve to zero address");
        
        _allowances[owner][spender] = amount;
        emit Approval(owner, spender, amount);
    }
    
    function _spendAllowance(address owner, address spender, uint256 amount) internal {
        uint256 currentAllowance = allowance(owner, spender);
        require(currentAllowance >= amount, "ERC20: insufficient allowance");
        
        _approve(owner, spender, currentAllowance - amount);
    }
    
    // ======================
    // MINTING & BURNING
    // ======================
    function _mint(address to, uint256 amount) internal {
        require(to != address(0), "MASIA: mint to zero address");
        
        totalSupply += amount;
        _balances[to] += amount;
        
        emit Transfer(address(0), to, amount);
        emit TokensMinted(to, amount, currentPhaseIndex);
    }
    
    function burn(uint256 amount) public {
        _burn(msg.sender, amount);
    }
    
    function _burn(address from, uint256 amount) internal {
        require(from != address(0), "MASIA: burn from zero address");
        require(_balances[from] >= amount, "MASIA: burn amount exceeds balance");
        
        _balances[from] -= amount;
        totalSupply -= amount;
        
        emit Transfer(from, address(0), amount);
    }
    
    // ======================
    // OWNER FUNCTIONS
    // ======================
    function lockTransfers(bool lock) public onlyOwner {
        transfersLocked = lock;
        emit TransferLockChanged(lock);
    }
    
    function recoverTokens(address tokenAddress, uint256 amount) public onlyOwner {
        require(tokenAddress != address(this), "MASIA: cannot recover self");
        // Implementation for recovering accidentally sent tokens
        // (Requires integration with token contract - simplified here)
        payable(owner).transfer(amount);
    }
    
    function withdrawStuckETH() public onlyOwner {
        payable(owner).transfer(address(this).balance);
    }
    
    // ======================
    // VIEW FUNCTIONS
    // ======================
    function getVestingSchedule(address holder) public view returns (
        uint256 startTime,
        uint256 cliffDuration,
        uint256 vestingDuration,
        uint256 totalAmount,
        uint256 releasedAmount,
        bool revocable,
        bool initialized
    ) {
        VestingSchedule memory schedule = vestingSchedules[holder];
        return (
            schedule.startTime,
            schedule.cliffDuration,
            schedule.vestingDuration,
            schedule.totalAmount,
            schedule.releasedAmount,
            schedule.revocable,
            schedule.initialized
        );
    }
    
    function getCurrentPhase() public view returns (uint256, string memory, uint256, uint256, bool) {
        if (currentPhaseIndex >= phases.length) {
            return (currentPhaseIndex, "Completed", 0, 0, false);
        }
        
        Phase memory current = phases[currentPhaseIndex];
        return (
            currentPhaseIndex,
            current.name,
            current.dilutionPercentage,
            current.endTime - block.timestamp,
            current.active
        );
    }
    
    function getPhase(uint256 index) public view returns (
        string memory name,
        uint256 startTime,
        uint256 endTime,
        uint256 dilutionPercentage,
        bool active
    ) {
        require(index < phases.length, "MASIA: phase does not exist");
        Phase memory phase = phases[index];
        return (
            phase.name,
            phase.startTime,
            phase.endTime,
            phase.dilutionPercentage,
            phase.active
        );
    }
